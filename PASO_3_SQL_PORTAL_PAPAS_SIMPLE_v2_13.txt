-- Ejecutar completo en Supabase > SQL Editor.
-- Usa tablas NUEVAS con sufijo _v213.
-- No borra jugadores, pagos ni tablas anteriores.
-- Esta versión prioriza funcionamiento estable para la academia.

create table if not exists public.parent_accounts_v213 (
  id uuid primary key default gen_random_uuid(),
  login text not null unique,
  display_name text not null,
  phone text,
  access_code text not null,
  active boolean not null default true,
  created_at timestamptz not null default now()
);

create table if not exists public.parent_player_links_v213 (
  id uuid primary key default gen_random_uuid(),
  parent_account_id uuid not null references public.parent_accounts_v213(id) on delete cascade,
  player_id text not null references public.players(id) on delete cascade,
  active boolean not null default true,
  created_at timestamptz not null default now(),
  unique(parent_account_id, player_id)
);

create table if not exists public.parent_sessions_v213 (
  token uuid primary key default gen_random_uuid(),
  parent_account_id uuid not null references public.parent_accounts_v213(id) on delete cascade,
  expires_at timestamptz not null default now() + interval '30 days',
  created_at timestamptz not null default now()
);

alter table public.parent_accounts_v213 enable row level security;
alter table public.parent_player_links_v213 enable row level security;
alter table public.parent_sessions_v213 enable row level security;

drop policy if exists "admin manage parent accounts v213" on public.parent_accounts_v213;
drop policy if exists "admin manage parent links v213" on public.parent_player_links_v213;
drop policy if exists "admin manage parent sessions v213" on public.parent_sessions_v213;

create policy "admin manage parent accounts v213"
on public.parent_accounts_v213
for all
to authenticated
using (true)
with check (true);

create policy "admin manage parent links v213"
on public.parent_player_links_v213
for all
to authenticated
using (true)
with check (true);

create policy "admin manage parent sessions v213"
on public.parent_sessions_v213
for all
to authenticated
using (true)
with check (true);

create or replace function public.ducks_parent_login_v213(
  p_login text,
  p_password text
)
returns table(ok boolean, message text, token uuid)
language plpgsql
security definer
set search_path = public
as $$
declare
  v_account public.parent_accounts_v213%rowtype;
  v_token uuid;
  v_input text;
begin
  v_input := lower(trim(coalesce(p_login,'')));

  select * into v_account
  from public.parent_accounts_v213
  where active = true
    and (
      lower(login) = v_input
      or regexp_replace(coalesce(login,''), '\D', '', 'g') = regexp_replace(coalesce(p_login,''), '\D', '', 'g')
      or lower(coalesce(phone,'')) = v_input
      or regexp_replace(coalesce(phone,''), '\D', '', 'g') = regexp_replace(coalesce(p_login,''), '\D', '', 'g')
    )
  limit 1;

  if not found then
    return query select false, 'Usuario o contraseña incorrectos.', null::uuid;
    return;
  end if;

  if trim(v_account.access_code) <> trim(coalesce(p_password,'')) then
    return query select false, 'Usuario o contraseña incorrectos.', null::uuid;
    return;
  end if;

  insert into public.parent_sessions_v213(parent_account_id)
  values (v_account.id)
  returning parent_sessions_v213.token into v_token;

  return query select true, 'OK', v_token;
end;
$$;

create or replace function public.ducks_parent_portal_v213(p_token uuid)
returns jsonb
language plpgsql
security definer
set search_path = public
as $$
declare
  v_account_id uuid;
  v_account jsonb;
  v_players jsonb;
  v_payments jsonb;
begin
  select s.parent_account_id into v_account_id
  from public.parent_sessions_v213 s
  where s.token = p_token and s.expires_at > now();

  if v_account_id is null then
    return jsonb_build_object('ok', false, 'message', 'Sesión inválida.');
  end if;

  select jsonb_build_object('id', id, 'login', login, 'display_name', display_name, 'phone', phone)
  into v_account
  from public.parent_accounts_v213
  where id = v_account_id;

  select coalesce(jsonb_agg(to_jsonb(p) order by p.name), '[]'::jsonb)
  into v_players
  from public.players p
  join public.parent_player_links_v213 l on l.player_id = p.id
  where l.parent_account_id = v_account_id and l.active = true;

  select coalesce(jsonb_agg(to_jsonb(pay) order by pay.created_at desc), '[]'::jsonb)
  into v_payments
  from public.payments pay
  join public.parent_player_links_v213 l on l.player_id = pay.player_id
  where l.parent_account_id = v_account_id and l.active = true;

  return jsonb_build_object('ok', true, 'account', v_account, 'players', v_players, 'payments', v_payments);
end;
$$;

create or replace function public.ducks_parent_submit_payment_v213(
  p_token uuid,
  p_player_id text,
  p_payment_date date,
  p_period text,
  p_amount numeric,
  p_method text,
  p_notes text,
  p_evidence_url text,
  p_evidence_name text
)
returns jsonb
language plpgsql
security definer
set search_path = public
as $$
declare
  v_account_id uuid;
  v_player_name text;
  v_login text;
begin
  select s.parent_account_id into v_account_id
  from public.parent_sessions_v213 s
  where s.token = p_token and s.expires_at > now();

  if v_account_id is null then
    return jsonb_build_object('ok', false, 'message', 'Sesión inválida.');
  end if;

  if not exists (
    select 1 from public.parent_player_links_v213
    where parent_account_id = v_account_id and player_id = p_player_id and active = true
  ) then
    return jsonb_build_object('ok', false, 'message', 'Jugador no autorizado para esta cuenta.');
  end if;

  select name into v_player_name from public.players where id = p_player_id;
  select login into v_login from public.parent_accounts_v213 where id = v_account_id;

  insert into public.payments(
    player_id, student_name, payment_date, period, amount, method, submitted_by,
    notes, confirmation_status, evidence_url, evidence_name
  )
  values (
    p_player_id, coalesce(v_player_name,''), p_payment_date, p_period, p_amount, p_method, v_login,
    p_notes, 'Pendiente de confirmación', p_evidence_url, p_evidence_name
  );

  return jsonb_build_object('ok', true);
end;
$$;

grant execute on function public.ducks_parent_login_v213(text,text) to anon, authenticated;
grant execute on function public.ducks_parent_portal_v213(uuid) to anon, authenticated;
grant execute on function public.ducks_parent_submit_payment_v213(uuid,text,date,text,numeric,text,text,text,text) to anon, authenticated;

select 'OK - Ducks CRM v2.13 instalado correctamente' as resultado;
