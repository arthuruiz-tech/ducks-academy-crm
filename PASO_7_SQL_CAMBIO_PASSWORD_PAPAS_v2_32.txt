-- Ducks CRM v2.32 - Cambio de contraseña desde Portal de Papás
create or replace function public.ducks_parent_change_password_v232(
  p_token uuid, p_new_password text
)
returns table(ok boolean, message text)
language plpgsql
security definer
set search_path=public
as $$
declare v_account_id uuid;
begin
  if length(trim(coalesce(p_new_password,'')))<4 then
    return query select false,'La contraseña debe tener al menos 4 caracteres.'; return;
  end if;
  select parent_account_id into v_account_id
  from public.parent_sessions_v213 where token=p_token limit 1;
  if v_account_id is null then
    return query select false,'Sesión no válida. Inicia sesión nuevamente.'; return;
  end if;
  update public.parent_accounts_v213 set access_code=trim(p_new_password) where id=v_account_id;
  return query select true,'Contraseña actualizada correctamente.';
end;
$$;
grant execute on function public.ducks_parent_change_password_v232(uuid,text) to anon,authenticated;
select 'OK - función cambio contraseña v2.32 creada' as resultado;
