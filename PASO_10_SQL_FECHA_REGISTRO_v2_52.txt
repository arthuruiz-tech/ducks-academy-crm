alter table public.players
  add column if not exists registration_date date;

update public.players
set registration_date = coalesce(
  (created_at at time zone 'America/Mexico_City')::date,
  (now() at time zone 'America/Mexico_City')::date
)
where registration_date is null;

alter table public.players
  alter column registration_date
  set default ((now() at time zone 'America/Mexico_City')::date);

alter table public.players
  alter column registration_date set not null;

update public.players
set payment_day = 5
where payment_day is null
   or payment_day < 1
   or payment_day > 31;

alter table public.players
  alter column payment_day set default 5;

alter table public.players
  alter column payment_day set not null;
